```swift
case CGEventType(rawValue: 0xFFFFFFFE)!, // tapDisabledByTimeout

    }
    return nil
```

Note: the existing `handleEvent` switch is on `type` with cases `.keyDown`, `.keyUp`, `.flagsChanged`, and `default`. The tap-disabled cases use force-unwrapped raw-value initializers because these CGEventType values are not enumerated in Swift. Add them before `default:`.

4. Add the health check methods after the existing `stop()` method:
```swift
func startHealthCheck() {
    healthTimer = Timer(timeInterval: 2.0, repeats: true) { [weak self] _ in
        self?.checkTapHealth()
    }
    RunLoop.main.add(healthTimer!, forMode: .common)
}

private func checkTapHealth() {
    guard let tap = eventTap else {
        DispatchQueue.main.async { self.isTapHealthy = false }
        return
    }
    let enabled = CGEvent.tapIsEnabled(tap: tap)
    if !enabled {
        CGEvent.tapEnable(tap: tap, enable: true)
    }
    let nowHealthy = CGEvent.tapIsEnabled(tap: tap)
    DispatchQueue.main.async { self.isTapHealthy = nowHealthy }
}

func stopHealthCheck() {
    healthTimer?.invalidate()
    healthTimer = nil
}
```

5. At the end of the `start()` method (after `CGEvent.tapEnable(tap: eventTap, enable: true)` on line 93), add:
```swift
startHealthCheck()
```

6. In the `stop()` method, add `stopHealthCheck()` as the first line.

7. In `deinit`, add `stopHealthCheck()` before the existing `stop()` call.

```swift
    }
}
```

This design means: when unhealthy, the icon shows a warning triangle regardless of phase. When health recovers, it restores the normal phase-driven icon.
